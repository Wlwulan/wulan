const SafeRequest = require("../utils/SafeRequest");
/**
 * Index类 获取后台有关于图书相关数据类
 * @class
 */
 class Index{
    /**
     * @constructor
     * @param {string} app KOA2执行的上下文环境 
     */
     constructor(app){}
     /**
      * 获取后台的全部图书数据方法
      * @param {*} url 接口API
      * @param {*} options  配置项
      * @example
      * return new Promise
      * getData(url, options)
      */
     getData(url, options){
        const safeRequest = new SafeRequest("books/index", options);
        return safeRequest.fetch();
     }
 }
 module.exports = Index;
